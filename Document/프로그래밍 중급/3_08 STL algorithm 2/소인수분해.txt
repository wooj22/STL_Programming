void prime_factor(int a) {
	cout << a << "의 소인수 분해 : ";
	for (int i = 2; i <= a; i++) {
		while (a % i == 0) {
			a /= i;
			cout << i << " ";
		}
	}
}

1. 최대 공약수

int gcd(int a, int b) {
	if (a % b == 0) {
		return b;
	}
	else {
		return gcd(b, a % b);
	}
}

int gcd2(int a, int b) {
	while (b != 0) {
		int temp = a % b;
		a = b;
		b = temp;
	}
	return a;
}

2. 최소 공배수 - 간단하게 서로를 곱한 것에 최대 공약수를 나눠주면 된다.

int lcm(int a, int b) {
	return a * b / gcd(a, b);
}
